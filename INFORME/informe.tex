\begin{lstlisting}[language=Python]
  import numpy as np
  from scipy.integrate import simps

  # Calcular el area bajo la curva usando integracion numerica (Simpson)
  area = simps(y, x)

  # Centroide en x
  x_bar = simps(x * y, x) / area
